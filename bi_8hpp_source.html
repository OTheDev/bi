346void bi_t::init_one_digit(bi_t& x, T value) {
347 x.resize_(1);

358template <std::integral T>
359void bi_t::init_atleast_one_digit(bi_t& x, T value) {
360 using UnsignedT = typename std::make_unsigned<T>::type;
361
362 size_t size = 1;
363 UnsignedT temp, uvalue;
364
365 if (value >= 0) {
366 uvalue = value;
367 x.negative_ = false;
368 } else {
369 uvalue = static_cast<UnsignedT>(0) - static_cast<UnsignedT>(value);
370 x.negative_ = true;
371 }
372
373 temp = uvalue;
374
375 while (temp >>= bi_dwidth) {
376 ++size;
377 }
378
379 x.resize_(size);
380
381 for (size_t i = 0; uvalue != 0; ++i) {
382 x[i] = static_cast<digit>(uvalue);
383 uvalue >>= bi_dwidth;
384 }
385}

390template <std::integral T>
391bi_t::bi_t(T value) {
392 using UnsignedT = typename std::make_unsigned<T>::type;
393
394 if (value == 0) {

396 negative_ = false;
397 return;
398 }
399
400 if constexpr (std::numeric_limits<T>::max() <= bi_dmax) {
401 bi_t::init_one_digit(*this, value);
402 } else {
403 bi_t::init_atleast_one_digit(*this, value);

426template <std::integral T>
427bi_t& bi_t::operator=(T value) {
428 using UnsignedT = typename std::make_unsigned<T>::type;
429
430 UnsignedT uvalue;
431 if (value == 0) {
432 resize_(0);
433 negative_ = false;
434 return *this;
435 } else if (value < 0) {
436 uvalue = static_cast<UnsignedT>(0) - static_cast<UnsignedT>(value);
437 negative_ = true;
438 } else {
439 uvalue = value;
440 negative_ = false;
441 }
442
443 if constexpr (std::numeric_limits<T>::max() <= bi_dmax) {
444 resize_(1);
445 vec_[0] = static_cast<digit>(uvalue);
446 } else {
447 UnsignedT temp = uvalue;
448
449 size_t n_digits = 1;
450 while (temp >>= bi_dwidth) {
451 ++n_digits;
452 }
453
454 resize_(n_digits);
455 for (size_t i = 0; uvalue != 0; ++i) {
456 vec_[i] = static_cast<digit>(uvalue);
457 uvalue >>= bi_dwidth;
458 }
459 }
460
461 return *this;
462}

466template <std::integral T>
467int bi_t::cmp(const bi_t& a, T b) noexcept {
468 if (b == 0) {
469 if (a.size() == 0) {
470 return 0; // a
471 }
472 return a.negative() ? -1 : 1; // b
473 }
474
475 using UnsignedT = typename std::make_unsigned<T>::type;
476 const bool b_negative = b < 0;
477 const UnsignedT unsigned_b =
478 b_negative ? -static_cast<UnsignedT>(b) : static_cast<UnsignedT>(b);
479
480 if (a.negative() && !b_negative) {
481 return -1; // c
482 }
483 if (!a.negative() && b_negative) {
484 return 1; // d
485 }
486
487 size_t n_b_digits = 0;
488 if constexpr (std::numeric_limits<UnsignedT>::max() <= bi_dmax) {
489 n_b_digits = (unsigned_b != 0) ? 1 : 0;
490 } else {
491 UnsignedT temp_b = unsigned_b;
492 while (temp_b != 0) {
493 temp_b >>= bi_dwidth;
494 n_b_digits++;
495 }
496 }
497
498 const size_t a_size = a.size();
499 if (a_size < n_b_digits) {
500 return a.negative() ? 1 : -1; // e
501 }
502
503 if (a_size > n_b_digits) {
504 return a.negative() ? -1 : 1; // f
505 }
506
507 for (size_t i = n_b_digits; i-- > 0;) {
508 const digit a_digit = a[i];
509 const digit b_digit = static_cast<digit>(unsigned_b >> (bi_dwidth * i));
510
511 if (a_digit < b_digit) {
512 return a.negative() ? 1 : -1; // g
513 }
514 if (a_digit > b_digit) {
515 return a.negative() ? -1 : 1; // h
516 }
517 }
518 return 0; // i
519}

655inline digit_vector to_twos_complement(const digit_vector& vec) {
656 const size_t vec_size = vec.size();
657 digit_vector result;
658 result.resize(vec_size);
659
660 bool carry = true;
661 for (size_t i = 0; i < vec_size; ++i) {
662 const digit sum = ~vec[i] + carry;
663 carry = sum < static_cast<uint8_t>(carry);
664 result[i] = sum;
665 }
666
667 return result;
668}

683template <bi_t::BitwiseOperation Op>
684void bi_t::bitwise_operation_impl(bi_t& result, const bi_t& x, const bi_t& y) {
685 size_t max_size = std::max(x.size(), y.size());
686
687 const digit_vector& lhs_digits =
688 x.negative_ ? to_twos_complement(x.vec_) : x.vec_;
689 const digit_vector& rhs_digits =
690 y.negative_ ? to_twos_complement(y.vec_) : y.vec_;
691
692 bool result_negative; // NOLINT(cppcoreguidelines-init-variables)
693 if constexpr (Op == BitwiseOperation::AND) {
694 result_negative = x.negative_ && y.negative_;
695 } else if constexpr (Op == BitwiseOperation::OR) {
696 result_negative = x.negative_ || y.negative_;
697 } else if constexpr (Op == BitwiseOperation::XOR) {
698 result_negative = x.negative_ != y.negative_;
699 ++max_size;
700 }
701
702 result.resize_(max_size);
703
704 for (size_t i = 0; i < result.size(); ++i) {
705 const digit lhs_digit =
706 i < lhs_digits.size() ? lhs_digits[i] : (x.negative_ ? bi_dmax : 0);
707 const digit rhs_digit =
708 i < rhs_digits.size() ? rhs_digits[i] : (y.negative_ ? bi_dmax : 0);
709
710 if constexpr (Op == BitwiseOperation::AND) {
711 result[i] = lhs_digit & rhs_digit;
712 } else if constexpr (Op == BitwiseOperation::OR) {
713 result[i] = lhs_digit | rhs_digit;
714 } else if constexpr (Op == BitwiseOperation::XOR) {
715 result[i] = lhs_digit ^ rhs_digit;
716 }
717 }
718
719 result.negative_ = result_negative;
720
721 if (result.negative_) {
722 to_twos_complement_in_place(result.vec_);
723 }
724
725 result.trim_trailing_zeros();
726}
