105void h_::increment_abs(bi_t& x) {
106 if (x.size() == 0 || x[x.size() - 1] == std::numeric_limits<digit>::max()) {
107 x.reserve_(x.size() + 1);
108 }
109
110 if (x.size() == 0) {
111 x.resize_(1);
112 x[0] = 1;
113 return;
114 }
115
116 size_t i = 0;
117 bool carry = true;
118
119 while (carry && i < x.size()) {
120 if (x[i] == std::numeric_limits<digit>::max()) {
121 x[i] = 0;
122 } else {
123 ++x[i];
124 carry = false;
125 }
126 ++i;
127 }
128
129 if (carry) {
130 x.vec_.push_back(1);
131 }
132}

134void h_::decrement_abs(bi_t& x) {
135 if (x.size() == 0) {
136 x.resize_(1);
137 x[0] = 1;
138 x.negative_ = true;
139 return;
140 }
141
142 size_t i = 0;
143 bool borrow = true;
144
145 while (borrow && i < x.size()) {
146 if (x[i] == 0) {
147 // Borrow from the next more significant digit
148 x[i] = std::numeric_limits<digit>::max();
149 } else {
150 --x[i];
151 borrow = false;
152 }
153 ++i;
154 }
155
156 x.trim_trailing_zeros();
157}

223void h_::add_abs(bi_t& r, const bi_t& x, const bi_t& y) {
224 // Algorithm assumes x.size() >= y.size()
225 const bi_t& large = x.size() >= y.size() ? x : y;
226 const bi_t& small = x.size() >= y.size() ? y : x;
227
228 r.reserve_(large.size() + 1);
229
230 digit temp{};
231 bool carry = 0;
232 size_t i = 0;
233 for (; i < small.size(); ++i) {
234 uints::uaddc(temp, large[i], small[i], carry);
235 r[i] = temp;
236 }
237
238 while (i < large.size()) {
239 temp = large[i] + carry;
240 carry = temp < static_cast<uint8_t>(carry); // MSVC warning C4804
241 r[i++] = temp;
242 }
243
244 r[i] = carry;
245 r.resize_(large.size() + 1);
246 r.trim_trailing_zeros();
247 r.negative_ = false;
248}

251void h_::add(bi_t& r, const bi_t& x, const bi_t& y) {
252 if (x.negative()) {
253 if (y.negative()) {
254 // x < 0, y < 0 ==> x = -|x|, y = -|y|. ==> x + y = -(|x| + |y|)
255 add_abs(r, x, y);
256 r.negative_ = true;
257 } else {
258 // x < 0, y >= 0 ==> x = -|x|, y = |y| ==> x + y = |y| - |x|
259 sub_abs(r, y, x);
260 }
261 } else {
262 if (y.negative()) {
263 // x >= 0, y < 0 ==> x = |x|, y = -|y| ==> x + y = |x| - |y|
264 sub_abs(r, x, y);
265 } else {
266 // x >= 0, y >= 0 ==> x = |x|, y = |y| ==> x + y = |x| + |y|
267 add_abs(r, x, y);
268 }
269 }

273void h_::sub_abs_gt(bi_t& r, const bi_t& x, const bi_t& y) {
274 assert(x.size() >= y.size());
275
276 r.reserve_(x.size());
277
278 digit temp{};
279 bool borrow = 0;
280 size_t i = 0;
281 for (; i < y.size(); ++i) {
282 uints::usubb(temp, x[i], y[i], borrow);
283 r[i] = temp;
284 }
285 while (i < x.size()) {
286 temp = x[i] - borrow;
287 borrow = temp > x[i];
288 r[i++] = temp;
289 }
290
291 r.resize_(x.size());
292 r.trim_trailing_zeros();
293 r.negative_ = false;
294}

297void h_::sub_abs(bi_t& r, const bi_t& x, const bi_t& y) {
298 if (x.size() == y.size()) {
299 const auto [it_x, it_y] = std::mismatch(x.rbegin(), x.rend(), y.rbegin());
300
301 if (it_x == x.rend()) {
302 // x, y have the same digits (or both are zero)
303 r.resize_(0);
304 r.negative_ = false;
305 return;
306 }
307
308 if (*it_y > *it_x) {
309 sub_abs_gt(r, y, x);
310 r.negative_ = true;
311 } else {
312 sub_abs_gt(r, x, y);
313 }
314 } else if (x.size() > y.size()) {
315 sub_abs_gt(r, x, y);
316 } else {
317 sub_abs_gt(r, y, x);
318 r.negative_ = true;
319 }
320}

323void h_::sub(bi_t& r, const bi_t& x, const bi_t& y) {
324 if (x.negative()) {
325 if (y.negative()) {
326 // x < 0, y < 0 ==> x = -|x|, y = -|y| ==> x - y = |y| - |x|
327 sub_abs(r, y, x);
328 } else {
329 // x < 0, y >= 0 ==> x = -|x|, y = |y| ==> x - y = -(|x| + |y|)
330 add_abs(r, x, y);
331 r.negative_ = true;
332 }
333 } else {
334 if (y.negative()) {
335 // x >= 0, y < 0 ==> x = |x|, y = -|y| ==> x - y = |x| + |y|
336 add_abs(r, x, y);
337 } else {
338 // x >= 0, y >= 0 ==> x = |x|, y = |y| ==> x - y = |x| - |y|
339 sub_abs(r, x, y);
340 }
341 }

391inline void h_::imul1add1(bi_t& x, digit v, digit k = 0) {
392 for (auto& d : x.vec_) {
393 const ddigit t = static_cast<ddigit>(d) * v + k;
394 k = t >> bi_dwidth; // floor(t / 2^{bi_dwidth})
395 d = static_cast<digit>(t); // t mod 2^{bi_dwidth}
396 }
397
398 if (k) {
399 x.vec_.push_back(k);
400 }

415void h_::mul_algo_square(bi_t& w, const bi_t& a, const size_t t) {
416 std::fill(w.begin(), w.end(), 0);
417
418 for (size_t i = 0; i < t; ++i) {
419 ddigit uv = w[2 * i] + static_cast<ddigit>(a[i]) * a[i];
420 w[2 * i] = static_cast<digit>(uv); // set w[2 * i] <- v
421 ddigit c = uv >> bi_dwidth; // set c <- u
422
423 for (size_t j = i + 1; j < t; ++j) {
424 // (hi, lo) represents a quadruple digit
425 ddigit hi{0};
426 ddigit lo = static_cast<ddigit>(a[j]) * a[i];
427
428 // Multiply product by two
429 hi = lo >> (std::numeric_limits<ddigit>::digits - 1);
430 lo <<= 1;
431
432 // Now add w[i + j] and c to (hi, lo)
433 hi += uints::uadd_overflow(lo, lo, static_cast<ddigit>(w[i + j]));
434 hi += uints::uadd_overflow(lo, lo, c);
435
436 w[i + j] = static_cast<digit>(lo); // set w[i + j] <- v
437 c = (hi << bi_dwidth) | (lo >> bi_dwidth); // set c <- u
438 }
439
440 for (size_t k = i + t; c > 0; ++k) {
441 c += w[k];
442 w[k] = static_cast<digit>(c);
443 c >>= bi_dwidth;
444 }
445 }
446}

448void h_::mul_algo_knuth(bi_t& w, const bi_t& a, const bi_t& b, const size_t m,
449 const size_t n) {
450 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
451 std::fill(w.begin(), w.begin() + m, 0);
452
453 for (size_t j = 0; j < n; ++j) {
454 digit k = 0;
455 for (size_t i = 0; i < m; ++i) {
456 const ddigit t =
457 static_cast<ddigit>(a[i]) * b[j] + static_cast<ddigit>(w[i + j]) + k;
458 k = t >> bi_dwidth; // floor(t / 2^{bi_dwidth})
459 w[i + j] = static_cast<digit>(t); // t mod 2^{bi_dwidth}
460 }
461 w[j + m] = k;
462 }
463}

506void h_::mul_karatsuba(bi_t& w, const bi_t& u, const bi_t& v) {
507 const size_t n = std::min(u.size(), v.size()) >> 1;
508
509 bi_t u0, u1, v0, v1;
510 bisect(u, u0, u1, n);
511 bisect(v, v0, v1, n);
512
513 bi_t a, b, c;
514 mul(a, u1, v1); // a = u1 * v1
515 mul(b, u0, v0); // b = u0 * v0
516
517 u1 += u0;
518 v1 += v0;
519 mul(c, u1, v1);
520 c -= a + b; // c = (u1 + u0)(v1 + v0) - (u0v0 + u1v1)
521
522 a <<= (2 * n * bi_dbits);
523 c <<= (n * bi_dbits);
524 w = a + c + b; // w = (bi_base ** 2n) * a + (bi_base ** n) * c + b
525}

534void h_::mul_standard(bi_t& result, const bi_t& a, const bi_t& b) {
535 if (a.size() == 0 || b.size() == 0) {
536 result.resize_(0);
537 result.negative_ = false;
538 return;
539 }
540
541 const size_t m = a.size();
542 const size_t n = b.size();
543
544 const bool overlap = &result == &a || &result == &b;
545
546 bi_t temp;
547 bi_t& target = overlap ? temp : result;
548
549 target.resize_(m + n);
550
551 if (&a != &b) {
552 h_::mul_algo_knuth(target, a, b, m, n);
553 } else {
554 h_::mul_algo_square(target, a, m);
555 }
556
557 target.trim_trailing_zeros();
558
559 if (overlap) {
560 result.swap(target);
561 }
562}

604digit h_::div_algo_digit(bi_t& q, const bi_t& u, digit v) noexcept {
605 ddigit rem{0};
606
607 for (size_t j = u.size() - 1; j < std::numeric_limits<size_t>::max(); --j) {
608 const ddigit temp = (rem << bi_dwidth) | u[j]; // rem * bi_base + u[j]
609 q[j] = temp / v;
610 rem = temp % v;
611 }
612
613 q.trim_trailing_zeros();
614 return static_cast<digit>(rem);
615}

660void h_::div_algo_binary(bi_t& q, bi_t& r, const bi_t& u, const bi_t& v) {
661 // (2)
662 q = 0;
663 r = 0;
664
665 // (3)
666 for (unsigned long i = u.bit_length() - 1; i < ULONG_MAX; --i) {
667 // (3)(I)
668 r <<= 1;
669
670 // (3)(II)
671 if (u.test_bit(i)) {
672 r.set_bit(0);
673 }
674
675 // (3)(III)
676 if (h_::cmp_abs(r, v) >= 0) {
677 // (3)(III)(a)
678 h_::sub_abs(r, r, v);
679 // (3)(III)(b)
680 q.set_bit(i);
681 }
682 }

747void h_::div_algo_knuth(bi_t& q, bi_t& r, const bi_t& u, const bi_t& v) {
748#if defined(BI_DIGIT_64_BIT)
749 using sddigit = __int128;
750#else
751 using sddigit = int64_t;
752#endif
753
754 const size_t m = u.size();
755 const size_t n = v.size();
756
757 /* (1) Normalize */
758 // Find e such that b > 2^{e} * v_{n-1} >= b / 2
759 constexpr digit b_half = bi_base >> 1;
760 const digit v_msd = v[n - 1];
761 int e = 0;
762 while ((v_msd << e) < b_half) {
763 ++e;
764 }
765 const unsigned compl_e = bi_dwidth - e;
766
767 bi_t u_norm, v_norm;
768 u_norm.resize_(m + 1);
769 v_norm.resize_(n);
770
771 // u_norm set to u * 2^{e}. Cast to ddigit as e may be 0 which would be UB
772 u_norm[0] = u[0] << e;
773 for (size_t i = 1; i < m; ++i) {
774 u_norm[i] = (u[i] << e) | (static_cast<ddigit>(u[i - 1]) >> compl_e);
775 }
776 u_norm[m] = static_cast<ddigit>(u[m - 1]) >> compl_e;
777
778 // v_norm set to v * 2^{e}
779 v_norm[0] = v[0] << e;
780 for (size_t i = 1; i < n; ++i) {
781 v_norm[i] = (v[i] << e) | (static_cast<ddigit>(v[i - 1]) >> compl_e);
782 }
783
784 const digit vp = v_norm[n - 1], vpp = v_norm[n - 2];
785 /* (2) Initialize j. Also (7) Loop on j */
786 for (size_t j = m - n; j < std::numeric_limits<size_t>::max(); --j) {
787 /* (3) Calculate q_hat */
788 const ddigit tmp = u_norm[j + n] * bi_base + u_norm[j + n - 1];
789 ddigit q_hat = tmp / vp;
790 ddigit r_hat = tmp % vp;
791
792 while (q_hat == bi_base ||
793 q_hat * vpp > bi_base * r_hat + u_norm[j + n - 2]) {
794 --q_hat;
795 r_hat += vp;
796 if (r_hat >= bi_base)
797 break;
798 }
799
800 /* (4) Multiply and subtract */
801 sddigit b{};
802 for (size_t i = 0; i < n; ++i) {
803 const ddigit product = q_hat * v_norm[i];
804 const sddigit stmp =
805 static_cast<sddigit>(u_norm[j + i]) - static_cast<digit>(product) - b;
806 u_norm[j + i] = static_cast<digit>(stmp);
807 b = static_cast<sddigit>(product >> bi_dbits) - (stmp >> bi_dbits);
808 }
809 const bool neg{u_norm[j + n] < b};
810 u_norm[j + n] = static_cast<digit>(u_norm[j + n] - b);
811
812 /* (5) Test remainder */
813 q[j] = q_hat;
814
815 if (neg) {
816 /* (6) Add back */
817 // Decrease q_{j} by 1
818 --q[j];
819
820 // Add (0v_{n-1}...v_{0})_{b} to (u_{j+n}...u_{j})_{b}
821 digit c = 0;
822 for (size_t i = 0; i < n; ++i) {
823 const ddigit tmp = (static_cast<ddigit>(u_norm[j + i]) + v_norm[i]) + c;
824 u_norm[j + i] = tmp;
825 c = tmp >> bi_dwidth;
826 }
827 // A carry will occur to the left of u_{j+n} and it should be ignored
828 u_norm[j + n] += c;
829 assert(u_norm[j + n] == 0);
830 }
831 }
832
833 /* (8) Unnormalize */
834 // (u_{n-1}...u_{0})_{b} / 2^e
835 const size_t last_idx = n - 1;
836 r[last_idx] = u_norm[last_idx] >> e;
837 for (size_t i = last_idx; i-- > 0;) {
838 r[i] = (static_cast<ddigit>(u_norm[i + 1]) << compl_e) | (u_norm[i] >> e);
839 }
840
841 q.trim_trailing_zeros();
842 r.trim_trailing_zeros();
843}

858void h_::divide(bi_t& Q, bi_t& R, const bi_t& N, const bi_t& D) {
859 if (D.size() == 0) {
860 throw division_by_zero("Division by zero attempt.");
861 }
862
863 const size_t size_N = N.size();
864 const size_t size_D = D.size();
865
866 // |N| < |D| case
867 if ((size_N == size_D && N[size_N - 1] < D[size_D - 1]) || size_N < size_D) {
868 Q = 0; // |N| < |D| ==> |N| / |D| < 1 ==> Q computes to 0
869 R = N; // Computation (a/b)*b + a%b should equal a ==> a%b is a
870 return;
871 }
872
873 // TRUE: size_N >= size_D > 0
874
875 // Ensure enough space in Q and R
876 const size_t size_Q = size_N - size_D + 1;
877 Q.resize_(size_Q);
878 R.resize_(size_D);
879
880 // Unsigned integer division algorithms
881 if (size_D == 1) {
882 // Knuth (Vol. 2, 4.3.1, p. 272) recommends using the algorithm used in
883 // div_algo_single() when size_D is 1.
884 div_algo_single(Q, R, N, D);
885 } else {
886 div_algo_knuth(Q, R, N, D);
887 }
888
889 Q.negative_ = D.negative() ^ N.negative();
890 R.negative_ = R.size() > 0 && N.negative();
891}

906void h_::left_shift(bi_t& result, const bi_t& x, bi_bitcount_t n_bits) {
907 if (x.size() == 0) {
908 result.resize_unsafe_(0);
909 result.negative_ = false;
910 return;
911 }
912
913 if (n_bits == 0) {
914 result = x; // Copy assignment avoids copying if &result == &x
915 return;
916 }
917
918 const size_t size_x = x.size();
919 const bi_bitcount_t digit_shift = n_bits / bi_dwidth;
920 const unsigned bit_shift = n_bits % bi_dwidth;
921
922 const size_t size_result = size_x + digit_shift + (bit_shift ? 1 : 0);
923 if (size_result < size_x) {
924 throw overflow_error("Result size exceeds SIZE_MAX.");
925 }
926
927 // Create a temporary vector if `result` and `x` overlap
928 dvector temp_vec;
929 dvector& target_vec = (&result == &x) ? temp_vec : result.vec_;
930
931 target_vec.resize(size_result);
932
933 size_t i = 0;
934 // Zero-fill vacated digits
935 for (; i < digit_shift; ++i) {
936 target_vec[i] = 0;
937 }
938
939 const auto compl_bit_shift = bi_dwidth - bit_shift;
940

942 for (size_t j = 0; j < size_x; ++j, ++i) {
943 const digit current = x[j];
944 target_vec[i] = (current << bit_shift) | carry;
945 carry = bit_shift ? (current >> compl_bit_shift) : 0;
946 }
947
948 if (bit_shift) {
949 target_vec[i] = carry;
950 }
951
952 if (&result == &x) {
953 result.vec_ = std::move(temp_vec);
954 }
955
956 result.trim_trailing_zeros();
957 result.negative_ = x.negative();

968void h_::right_shift(bi_t& result, const bi_t& x, bi_bitcount_t n_bits) {
969 // TODO: maybe set `result = x` for special cases x.size() == 0 || n_bits == 0
970 const size_t size_x = x.size();
971
972 const bi_bitcount_t digit_shift = n_bits / bi_dwidth;
973 const unsigned bit_shift = n_bits % bi_dwidth;
974
975 if (size_x <= digit_shift) {
976 if (x.negative()) {
977 // Set result to -1
978 result.resize_(1);
979 result[0] = 1;
980 result.negative_ = true;
981 } else {
982 // Set result to 0
983 result.resize_unsafe_(0);
984 result.negative_ = false;
985 }
986 return;
987 }
988
989 result.resize_(size_x - digit_shift);
990
991 if (bit_shift == 0) {
992 for (size_t i = 0; i < result.size(); ++i) {
993 result[i] = x[i + digit_shift];
994 }
995 } else {
996 const unsigned compl_bit_shift = bi_dwidth - bit_shift;
997
998 for (size_t i = 0, k = digit_shift + 1; i < result.size(); ++i, ++k) {
999 result[i] = x[k - 1] >> bit_shift;
1000 if (k < size_x) {
1001 result[i] |= x[k] << compl_bit_shift;
1002 }
1003 }
1004 }
1005
1006 result.trim_trailing_zeros();
1007 result.negative_ = x.negative();
1008

1010
1011 // Adjust for floor division for negative numbers
1012 if (x.negative()) {
1013 bool subtract_one = false;
1014 if (bit_shift > 0) {
1015 const digit mask = (static_cast<digit>(1) << bit_shift) - 1;
1016 if ((x[digit_shift] & mask) != 0) {
1017 subtract_one = true;
1018 }
1019 }
1020 if (!subtract_one && digit_shift > 0) {
1021 for (size_t i = 0; i < digit_shift; ++i) {
1022 if (x[i] != 0) {
1023 subtract_one = true;
1024 break;
1025 }
1026 }
1027 }
1028 if (subtract_one) {

1030 }

1066 } else if constexpr (Op == BitwiseOperation::XOR) {

1071 result.negative_ = result_negative;
