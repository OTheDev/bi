90void h_::increment_abs(bi_t& x) {
91 if (x.size() == 0 || x[x.size() - 1] == std::numeric_limits<digit>::max()) {
92 x.reserve_(x.size() + 1);
93 }
94
95 if (x.size() == 0) {
96 x.resize_(1);
97 x[0] = 1;
98 return;
99 }
100
101 size_t i = 0;
102 bool carry = true;
103
104 while (carry && i < x.size()) {
105 if (x[i] == std::numeric_limits<digit>::max()) {
106 x[i] = 0;
107 } else {
108 ++x[i];
109 carry = false;
110 }
111 ++i;
112 }
113
114 if (carry) {
115 x.vec_.push_back(1);
116 }
117}

119void h_::decrement_abs(bi_t& x) {
120 if (x.size() == 0) {
121 x.resize_(1);
122 x[0] = 1;
123 x.negative_ = true;
124 return;
125 }
126
127 size_t i = 0;
128 bool borrow = true;
129
130 while (borrow && i < x.size()) {
131 if (x[i] == 0) {
132 // Borrow from the next more significant digit
133 x[i] = std::numeric_limits<digit>::max();
134 } else {
135 --x[i];
136 borrow = false;
137 }
138 ++i;
139 }
140
141 x.trim_trailing_zeros();
142}

208void h_::add_abs(bi_t& r, const bi_t& x, const bi_t& y) {
209 // Algorithm assumes x.size() >= y.size()
210 const bi_t& large = x.size() >= y.size() ? x : y;
211 const bi_t& small = x.size() >= y.size() ? y : x;
212
213 r.reserve_(large.size() + 1);
214
215 digit temp{};
216 bool carry = 0;
217 size_t i = 0;
218 for (; i < small.size(); ++i) {
219 uints::uaddc(temp, large[i], small[i], carry);
220 r[i] = temp;
221 }
222
223 while (i < large.size()) {
224 temp = large[i] + carry;
225 carry = temp < static_cast<uint8_t>(carry); // MSVC warning C4804
226 r[i++] = temp;
227 }
228
229 r[i] = carry;
230 r.resize_(large.size() + 1);
231 r.trim_trailing_zeros();
232 r.negative_ = false;
233}

236void h_::add(bi_t& r, const bi_t& x, const bi_t& y) {
237 if (x.negative()) {
238 if (y.negative()) {
239 // x < 0, y < 0 ==> x = -|x|, y = -|y|. ==> x + y = -(|x| + |y|)
240 add_abs(r, x, y);
241 r.negative_ = true;
242 } else {
243 // x < 0, y >= 0 ==> x = -|x|, y = |y| ==> x + y = |y| - |x|
244 sub_abs(r, y, x);
245 }
246 } else {
247 if (y.negative()) {
248 // x >= 0, y < 0 ==> x = |x|, y = -|y| ==> x + y = |x| - |y|
249 sub_abs(r, x, y);
250 } else {
251 // x >= 0, y >= 0 ==> x = |x|, y = |y| ==> x + y = |x| + |y|
252 add_abs(r, x, y);
253 }
254 }

258void h_::sub_abs_gt(bi_t& r, const bi_t& x, const bi_t& y) {
259 assert(x.size() >= y.size());
260
261 r.reserve_(x.size());
262
263 digit temp{};
264 bool borrow = 0;
265 size_t i = 0;
266 for (; i < y.size(); ++i) {
267 uints::usubb(temp, x[i], y[i], borrow);
268 r[i] = temp;
269 }
270 while (i < x.size()) {
271 temp = x[i] - borrow;
272 borrow = temp > x[i];
273 r[i++] = temp;
274 }
275
276 r.resize_(x.size());
277 r.trim_trailing_zeros();
278 r.negative_ = false;
279}

282void h_::sub_abs(bi_t& r, const bi_t& x, const bi_t& y) {
283 if (x.size() == y.size()) {
284 const auto [it_x, it_y] = std::mismatch(x.rbegin(), x.rend(), y.rbegin());
285
286 if (it_x == x.rend()) {
287 // x, y have the same digits (or both are zero)
288 r.resize_(0);
289 r.negative_ = false;
290 return;
291 }
292
293 if (*it_y > *it_x) {
294 sub_abs_gt(r, y, x);
295 r.negative_ = true;
296 } else {
297 sub_abs_gt(r, x, y);
298 }
299 } else if (x.size() > y.size()) {
300 sub_abs_gt(r, x, y);
301 } else {
302 sub_abs_gt(r, y, x);
303 r.negative_ = true;
304 }
305}

308void h_::sub(bi_t& r, const bi_t& x, const bi_t& y) {
309 if (x.negative()) {
310 if (y.negative()) {
311 // x < 0, y < 0 ==> x = -|x|, y = -|y| ==> x - y = |y| - |x|
312 sub_abs(r, y, x);
313 } else {
314 // x < 0, y >= 0 ==> x = -|x|, y = |y| ==> x - y = -(|x| + |y|)
315 add_abs(r, x, y);
316 r.negative_ = true;
317 }
318 } else {
319 if (y.negative()) {
320 // x >= 0, y < 0 ==> x = |x|, y = -|y| ==> x - y = |x| + |y|
321 add_abs(r, x, y);
322 } else {
323 // x >= 0, y >= 0 ==> x = |x|, y = |y| ==> x - y = |x| - |y|
324 sub_abs(r, x, y);
325 }
326 }

376inline void h_::imul1add1(bi_t& x, digit v, digit k = 0) {
377 for (auto& d : x.vec_) {
378 const ddigit t = static_cast<ddigit>(d) * v + k;
379 k = t >> bi_dwidth; // floor(t / 2^{bi_dwidth})
380 d = static_cast<digit>(t); // t mod 2^{bi_dwidth}
381 }
382
383 if (k) {
384 x.vec_.push_back(k);
385 }

394void h_::mul(bi_t& result, const bi_t& a, const bi_t& b) {
395 if (a.size() == 0 || b.size() == 0) {
396 result.resize_(0);
397 result.negative_ = false;
398 return;
399 }
400
401 const size_t m = a.size();
402 const size_t n = b.size();
403
404 const auto [n_result_digits, overflow] = uints::uadd_overflow(m, n);
405 if (overflow) {
406 throw overflow_error("");
407 }
408
409 const bool overlap = &result == &a || &result == &b;
410
411 bi_t temp;
412 bi_t& target = overlap ? temp : result;
413
414 target.resize_(n_result_digits);
415
416 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
417 std::fill(target.begin(), target.begin() + m, 0);
418
419 for (size_t j = 0; j < n; ++j) {
420 digit k = 0;
421 for (size_t i = 0; i < m; ++i) {
422 const ddigit t = static_cast<ddigit>(a[i]) * b[j] +
423 static_cast<ddigit>(target[i + j]) + k;
424 k = t >> bi_dwidth; // floor(t / 2^{bi_dwidth})
425 target[i + j] = static_cast<digit>(t); // t mod 2^{bi_dwidth}
426 }
427 target[j + m] = k;
428 }
429
430 target.trim_trailing_zeros();
431 target.negative_ = a.negative() != b.negative();
432
433 if (overlap) {
434 result.swap(target);
435 }
436}

468void h_::div_algo_single(bi_t& q, bi_t& r, const bi_t& u,
469 const bi_t& v) noexcept {
470 ddigit rem{0};
471
472 for (size_t j = u.size() - 1; j < std::numeric_limits<size_t>::max(); --j) {
473 const ddigit temp = (rem << bi_dwidth) | u[j]; // rem * bi_base + u[j]
474 q[j] = temp / v[0];
475 rem = temp % v[0];
476 }
477
478 r[0] = static_cast<digit>(rem);
479
480 q.trim_trailing_zeros();
481 r.trim_trailing_zeros();
482}

511void h_::div_algo_binary(bi_t& q, bi_t& r, const bi_t& u, const bi_t& v) {
512 // (2)
513 q = 0;
514 r = 0;
515
516 // (3)
517 for (unsigned long i = u.bit_length() - 1; i < ULONG_MAX; --i) {
518 // (3)(I)
519 r <<= 1;
520
521 // (3)(II)
522 if (u.test_bit(i)) {
523 r.set_bit(0);
524 }
525
526 // (3)(III)
527 if (h_::cmp_abs(r, v) >= 0) {
528 // (3)(III)(a)
529 h_::sub_abs(r, r, v);
530 // (3)(III)(b)
531 q.set_bit(i);
532 }
533 }

598void h_::div_algo_knuth(bi_t& q, bi_t& r, const bi_t& u, const bi_t& v) {
599#if defined(BI_DIGIT_64_BIT)
600 using sddigit = __int128;
601#else
602 using sddigit = int64_t;
603#endif
604
605 const size_t m = u.size();
606 const size_t n = v.size();
607
608 /* (1) Normalize */
609 // Find e such that b > 2^{e} * v_{n-1} >= b / 2
610 constexpr digit b_half = bi_base >> 1;
611 const digit v_msd = v[n - 1];
612 int e = 0;
613 while ((v_msd << e) < b_half) {
614 ++e;
615 }
616 const unsigned compl_e = bi_dwidth - e;
617
618 bi_t u_norm, v_norm;
619 u_norm.resize_(m + 1);
620 v_norm.resize_(n);
621
622 // u_norm set to u * 2^{e}. Cast to ddigit as e may be 0 which would be UB
623 u_norm[0] = u[0] << e;
624 for (size_t i = 1; i < m; ++i) {
625 u_norm[i] = (u[i] << e) | (static_cast<ddigit>(u[i - 1]) >> compl_e);
626 }
627 u_norm[m] = static_cast<ddigit>(u[m - 1]) >> compl_e;
628
629 // v_norm set to v * 2^{e}
630 v_norm[0] = v[0] << e;
631 for (size_t i = 1; i < n; ++i) {
632 v_norm[i] = (v[i] << e) | (static_cast<ddigit>(v[i - 1]) >> compl_e);
633 }
634
635 const digit vp = v_norm[n - 1], vpp = v_norm[n - 2];
636 /* (2) Initialize j. Also (7) Loop on j */
637 for (size_t j = m - n; j < std::numeric_limits<size_t>::max(); --j) {
638 /* (3) Calculate q_hat */
639 const ddigit tmp = u_norm[j + n] * bi_base + u_norm[j + n - 1];
640 ddigit q_hat = tmp / vp;
641 ddigit r_hat = tmp % vp;
642
643 while (q_hat == bi_base ||
644 q_hat * vpp > bi_base * r_hat + u_norm[j + n - 2]) {
645 --q_hat;
646 r_hat += vp;
647 if (r_hat >= bi_base)
648 break;
649 }
650
651 /* (4) Multiply and subtract */
652 sddigit b{};
653 for (size_t i = 0; i < n; ++i) {
654 const ddigit product = q_hat * v_norm[i];
655 const sddigit stmp =
656 static_cast<sddigit>(u_norm[j + i]) - static_cast<digit>(product) - b;
657 u_norm[j + i] = static_cast<digit>(stmp);
658 b = static_cast<sddigit>(product >> bi_dbits) - (stmp >> bi_dbits);
659 }
660 const bool neg{u_norm[j + n] < b};
661 u_norm[j + n] = static_cast<digit>(u_norm[j + n] - b);
662
663 /* (5) Test remainder */
664 q[j] = q_hat;
665
666 if (neg) {
667 /* (6) Add back */
668 // Decrease q_{j} by 1
669 --q[j];
670
671 // Add (0v_{n-1}...v_{0})_{b} to (u_{j+n}...u_{j})_{b}
672 digit c = 0;
673 for (size_t i = 0; i < n; ++i) {
674 const ddigit tmp = (static_cast<ddigit>(u_norm[j + i]) + v_norm[i]) + c;
675 u_norm[j + i] = tmp;
676 c = tmp >> bi_dwidth;
677 }
678 // A carry will occur to the left of u_{j+n} and it should be ignored
679 u_norm[j + n] += c;
680 assert(u_norm[j + n] == 0);
681 }
682 }
683
684 /* (8) Unnormalize */
685 // (u_{n-1}...u_{0})_{b} / 2^e
686 const size_t last_idx = n - 1;
687 r[last_idx] = u_norm[last_idx] >> e;
688 for (size_t i = last_idx; i-- > 0;) {
689 r[i] = (static_cast<ddigit>(u_norm[i + 1]) << compl_e) | (u_norm[i] >> e);
690 }
691
692 q.trim_trailing_zeros();
693 r.trim_trailing_zeros();
694}

709void h_::divide(bi_t& Q, bi_t& R, const bi_t& N, const bi_t& D) {
710 if (D.size() == 0) {
711 throw division_by_zero("Division by zero attempt.");
712 }
713
714 const size_t size_N = N.size();
715 const size_t size_D = D.size();
716
717 // |N| < |D| case
718 if ((size_N == size_D && N[size_N - 1] < D[size_D - 1]) || size_N < size_D) {
719 Q = 0; // |N| < |D| ==> |N| / |D| < 1 ==> Q computes to 0
720 R = N; // Computation (a/b)*b + a%b should equal a ==> a%b is a
721 return;
722 }
723
724 // TRUE: size_N >= size_D > 0
725
726 // Ensure enough space in Q and R
727 const size_t size_Q = size_N - size_D + 1;
728 Q.resize_(size_Q);
729 R.resize_(size_D);
730
731 // Unsigned integer division algorithms
732 if (size_D == 1) {
733 // Knuth (Vol. 2, 4.3.1, p. 272) recommends using the algorithm used in
734 // div_algo_single() when size_D is 1.
735 div_algo_single(Q, R, N, D);
736 } else {
737 div_algo_knuth(Q, R, N, D);
738 }
739
740 Q.negative_ = D.negative() ^ N.negative();
741 R.negative_ = R.size() > 0 && N.negative();
742}

757void h_::left_shift(bi_t& result, const bi_t& x, bi_bitcount_t n_bits) {
758 if (x.size() == 0) {
759 result.resize_unsafe_(0);
760 result.negative_ = false;
761 return;
762 }
763
764 if (n_bits == 0) {
765 result = x; // Copy assignment avoids copying if &result == &x
766 return;
767 }
768
769 const size_t size_x = x.size();
770 const bi_bitcount_t digit_shift = n_bits / bi_dwidth;
771 const unsigned bit_shift = n_bits % bi_dwidth;
772
773 const size_t size_result = size_x + digit_shift + (bit_shift ? 1 : 0);
774 if (size_result < size_x) {
775 throw overflow_error("Result size exceeds SIZE_MAX.");
776 }
777
778 // Create a temporary vector if `result` and `x` overlap
779 dvector temp_vec;
780 dvector& target_vec = (&result == &x) ? temp_vec : result.vec_;
781
782 target_vec.resize(size_result);
783
784 size_t i = 0;
785 // Zero-fill vacated digits
786 for (; i < digit_shift; ++i) {
787 target_vec[i] = 0;
788 }
789
790 const auto compl_bit_shift = bi_dwidth - bit_shift;
791
792 digit carry = 0;
793 for (size_t j = 0; j < size_x; ++j, ++i) {
794 const digit current = x[j];
795 target_vec[i] = (current << bit_shift) | carry;
796 carry = bit_shift ? (current >> compl_bit_shift) : 0;
797 }
798
799 if (bit_shift) {
800 target_vec[i] = carry;
801 }
802
803 if (&result == &x) {
804 result.vec_ = std::move(temp_vec);
805 }
806
807 result.trim_trailing_zeros();
808 result.negative_ = x.negative();
809}

819void h_::right_shift(bi_t& result, const bi_t& x, bi_bitcount_t n_bits) {
820 // TODO: maybe set `result = x` for special cases x.size() == 0 || n_bits == 0
821 const size_t size_x = x.size();
822
823 const bi_bitcount_t digit_shift = n_bits / bi_dwidth;
824 const unsigned bit_shift = n_bits % bi_dwidth;
825
826 if (size_x <= digit_shift) {
827 if (x.negative()) {
828 // Set result to -1
829 result.resize_(1);
830 result[0] = 1;
831 result.negative_ = true;
832 } else {
833 // Set result to 0
834 result.resize_unsafe_(0);
835 result.negative_ = false;
836 }
837 return;
838 }
839
840 result.resize_(size_x - digit_shift);
841
842 if (bit_shift == 0) {
843 for (size_t i = 0; i < result.size(); ++i) {
844 result[i] = x[i + digit_shift];
845 }
846 } else {
847 const unsigned compl_bit_shift = bi_dwidth - bit_shift;
848
849 for (size_t i = 0, k = digit_shift + 1; i < result.size(); ++i, ++k) {
850 result[i] = x[k - 1] >> bit_shift;
851 if (k < size_x) {
852 result[i] |= x[k] << compl_bit_shift;
853 }
854 }
855 }
856
857 result.trim_trailing_zeros();
858 result.negative_ = x.negative();
859
860 // At this point, the result is trunc(x / 2^{n_bits}) for all x
861
862 // Adjust for floor division for negative numbers
863 if (x.negative()) {
864 bool subtract_one = false;
865 if (bit_shift > 0) {
866 const digit mask = (static_cast<digit>(1) << bit_shift) - 1;
867 if ((x[digit_shift] & mask) != 0) {
868 subtract_one = true;
869 }
870 }
871 if (!subtract_one && digit_shift > 0) {
872 for (size_t i = 0; i < digit_shift; ++i) {
873 if (x[i] != 0) {
874 subtract_one = true;
875 break;
876 }
877 }
878 }
879 if (subtract_one) {
880 --result;
881 }
882 }

888template <h_::BitwiseOperation Op>
889void h_::bitwise_operation_impl(bi_t& result, const bi_t& x, const bi_t& y) {
890 size_t max_size = std::max(x.size(), y.size());
891
892 const dvector& lhs_digits = x.negative_ ? to_twos_complement(x.vec_) : x.vec_;
893 const dvector& rhs_digits = y.negative_ ? to_twos_complement(y.vec_) : y.vec_;
894
895 bool result_negative; // NOLINT(cppcoreguidelines-init-variables)
896 if constexpr (Op == BitwiseOperation::AND) {
897 result_negative = x.negative_ && y.negative_;
898 } else if constexpr (Op == BitwiseOperation::OR) {
899 result_negative = x.negative_ || y.negative_;
900 } else if constexpr (Op == BitwiseOperation::XOR) {
901 result_negative = x.negative_ != y.negative_;
902 ++max_size;
903 }
904
905 result.resize_(max_size);
906
907 for (size_t i = 0; i < result.size(); ++i) {
908 const digit lhs_digit =
909 i < lhs_digits.size() ? lhs_digits[i] : (x.negative_ ? bi_dmax : 0);
910 const digit rhs_digit =
911 i < rhs_digits.size() ? rhs_digits[i] : (y.negative_ ? bi_dmax : 0);
912
913 if constexpr (Op == BitwiseOperation::AND) {
914 result[i] = lhs_digit & rhs_digit;
915 } else if constexpr (Op == BitwiseOperation::OR) {
916 result[i] = lhs_digit | rhs_digit;
917 } else if constexpr (Op == BitwiseOperation::XOR) {
918 result[i] = lhs_digit ^ rhs_digit;
919 }
920 }
921
922 result.negative_ = result_negative;
923
924 if (result.negative_) {
925 to_twos_complement_in_place(result.vec_);
926 }
927
928 result.trim_trailing_zeros();
929}

932template <h_::BitwiseOperation Op>
933bi_t h_::bitwise_operation(const bi_t& a, const bi_t& b) {
934 bi_t result;

994template <std::integral T>
995int h_::cmp(const bi_t& a, T b) noexcept {
996 if (b == 0) {
997 if (a.size() == 0) {
998 return 0; // a
999 }
1000 return a.negative() ? -1 : 1; // b
1001 }
1002
1003 using UnsignedT = typename std::make_unsigned<T>::type;
1004 const bool b_negative = b < 0;
1005 const UnsignedT unsigned_b =
1006 b_negative ? -static_cast<UnsignedT>(b) : static_cast<UnsignedT>(b);
1007
1008 if (a.negative() && !b_negative) {
1009 return -1; // c
1010 }
1011 if (!a.negative() && b_negative) {
1012 return 1; // d
1013 }
1014
1015 size_t n_b_digits = 0;
1016 if constexpr (std::numeric_limits<UnsignedT>::max() <= bi_dmax) {
1017 n_b_digits = (unsigned_b != 0) ? 1 : 0;
1018 } else {
1019 UnsignedT temp_b = unsigned_b;
1020 while (temp_b != 0) {
1021 temp_b >>= bi_dwidth;
1022 n_b_digits++;
1023 }
1024 }
1025
1026 const size_t a_size = a.size();
1027 if (a_size < n_b_digits) {
1028 return a.negative() ? 1 : -1; // e
1029 }
1030
1031 if (a_size > n_b_digits) {
1032 return a.negative() ? -1 : 1; // f
1033 }
1034
1035 for (size_t i = n_b_digits; i-- > 0;) {
1036 const digit a_digit = a[i];
1037 const digit b_digit = static_cast<digit>(unsigned_b >> (bi_dwidth * i));
1038
1039 if (a_digit < b_digit) {
1040 return a.negative() ? 1 : -1; // g
1041 }
1042 if (a_digit > b_digit) {
1043 return a.negative() ? -1 : 1; // h
1044 }
1045 }
1046 return 0; // i
1047}

1441void h_::init_string(bi_t& x, const std::string& s) {
1442 // std::stoi and company allow leading whitespace and a plus/minus sign. We
1443 // follow suit.
1444
1445 // Allow leading whitespace
1446 auto it = std::find_if_not(s.begin(), s.end(),
1447 [](char ch) { return std::isspace(ch); });
1448
1449 // Allow plus/minus sign to precede first decimal digit
1450 x.negative_ = false;
1451 if (it != s.end()) {
1452 if (*it == '-') {
1453 x.negative_ = true;
1454 ++it;
1455 } else if (*it == '+') {
1456 ++it;
1457 }
1458 }
1459
1460 const auto start_digit = it;
1461 it = std::find_if_not(start_digit, s.end(),
1462 [](char ch) { return std::isdigit(ch); });
1463
1464 if (start_digit == it) {
1465 throw std::invalid_argument("Invalid string format.");
1466 }
1467
1468 size_t n_base10 = std::distance(start_digit, it); // it - start_digit
1469 const size_t n_digits = uints::div_ceil(n_base10, max_batch_size);
1470
1471 x.reserve_(n_digits);
1472 x.resize_unsafe_(0);
1473
1474 for (auto dec_it = start_digit; dec_it < it;) {
1475 /* We could replace all the code in the body of this loop with just
1476 * `imul1add1(10, *dec_it++ - '0');` and the end result will be the same.
1477 * However, it is more efficient to batch some base-10 digits together. */
1478 // Initialize batch value
1479 digit batch = 0;
1480
1481 // Calculate how many base-10 digits we can process in this batch
1482 const size_t remaining = std::distance(dec_it, it);
1483 const size_t digits_in_batch =
1484 std::min(remaining, static_cast<size_t>(max_batch_size));
1485
1486 // Convert batch substring to integer value
1487 for (size_t j = 0; j < digits_in_batch; ++j, ++dec_it) {
1488 // NOLINTNEXTLINE(cppcoreguidelines-avoid-magic-numbers)
1489 batch = (*dec_it - '0') + batch * 10;
1490 }
1491
1492 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
1493 h_::imul1add1(x, powers_of_ten[digits_in_batch], batch);
1494 }
1495
1496 x.trim_trailing_zeros();
1497}

1536 if (std::isnan(d) || std::isinf(d)) {
1537 throw from_float(
1538 "Conversion error: NaN or infinity cannot be converted to an integer.");
1539 }
1540
1541 if (d > -1 && d < 1) {
1542 x = 0;
1543 return;
1544 }
1545
1546 const bool neg = d < 0;
1547 if (neg) {
1548 d = -d;
1549 }
1550
1551 size_t n_digits{1};
1552 while (bi_base_dbl <= d) {
1553 // Equiv. to `d /= bi_base_dbl;`, but multiplication is generally cheaper
1554 d *= bi_base_dbl_reciprocal;

1558 x.resize_(n_digits);

1569int h_::cmp_abs(const bi_t& z, double dbl) noexcept {
1570 if (z.size() == 0) {
1571 return dbl > 0 ? -1 : 0;
1572 }
1573
1574 if (z.size() >= bi_cmp_dbl_size_upper) {
1575 return 1;
1576 }
1577
1578 // Same as dbl = std::ldexp(dbl, -static_cast<int>(bi_dbits) *
1579 // static_cast<int>(z.size() - 1));
1580 for (size_t j = 1; j < z.size(); ++j) {
1581 // Equiv. to `dbl /= bi_base_dbl;`, but multiplication is generally cheaper
1582 dbl *= bi_base_dbl_reciprocal;
1583 }
1584
1585 if (bi_base_dbl <= dbl) {
1586 return -1;
1587 }
1588
1589 for (auto it = z.rbegin(); it != z.rend(); ++it) {
1590 const digit cur = static_cast<digit>(dbl);
1591
1592 if (*it < cur) {
1593 return -1;
1594 } else if (*it > cur) {
1595 return 1;
1596 } else {
1597 dbl = (dbl - static_cast<double>(cur)) * bi_base_dbl;
1598 }
1599 }
1600
1601 return dbl > 0 ? -1 : 0;
1602}
